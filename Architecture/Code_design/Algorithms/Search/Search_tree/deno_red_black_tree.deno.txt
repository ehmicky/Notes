                                  #Left child < parent < right child
                                  #No self-balancing
                                  #FUNC(VAL, VAL2)->-1|0|1
                                  #  - defines order
                                  #  - also defines equality, e.g. for find()
                                  #  - can be descend (>) or ascend (def, <)

 ->VAL|null                       #
