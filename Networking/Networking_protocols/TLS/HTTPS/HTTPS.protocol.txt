MIXED CONTENT ==>                 #HTTPS needs to be enable on all website resources
                                  #  - including scripts, all pages, etc.
                                  #  - otherwise could attack on specific unsecure resources
                                  #This includes cookies
                                  #  - i.e. should use "secure" attribute
                                  #Solutions:
                                  #  - HSTS
                                  #  - HTTP -> HTTPS redirects, either:
                                  #     - permanent redirects
                                  #     - Link; rel="canonical" [S]
                                  #  - relative URIs

Strict-Transport-Security:        #HTTP Strict Transport Security (HSTS)
 max-age=NUM[; includeSubDomains] #Ask client to redirect HTTP users to HTTPS instead
 [S]                              #For NUM seconds
                                  #Not done if SSL certificate invalid
                                  #Prevents SSL striping
  - Must be supplied over HTTPS, so:
     - does not work if initial connection is HTTP, or in the time window when max-age expired.
     - However, some browsers include a preloaded list of websites using this header.
