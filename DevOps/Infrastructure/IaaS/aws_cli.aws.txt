JMESPATH ==>                  #Format of FILTER.
                              #Partial documentation:
                              #  - OBJ.VAR
                              #  - ARR[NUM]
                              #  - ARR[*]: all ARR members
                              #  - ARR[[NUM]:[NUM2]:[STEP]]:
                              #     - slice
                              #     - def: start, end, 1
                              #     - can be negative NUM (from end)
                              #  - ARR[*].ARR[]: like ARR[*].ARR[*], but flattened
                              #  - OBJ_ARR[?VAR==VAL]
                              #     - filter
                              #     - == can also be != <= < > >=
                              #     - VAL is `...` (STR) or `NUM`
                              #  - OBJ[_ARR].{NEW:OLD,...}:
                              #     - like OBJ.OLD, but rename to OBJ.NEW
                              #     - OLD can be FILTER
                              #  - OBJ[_ARR].[VAR,...]:
                              #     - each OBJ.VAR, as ARR
                              #     - VAR can be FILTER
                              #  - FILTER | FILTER2: pipe
                              #  - built-in FUNCs:
                              #     - abs|ceil|floor(NUM)->NUM
                              #     - avg|sum|min|max(NUM_ARR)->NUM
                              #     - contains(ARR|STR, STR2)->BOOL
                              #     - starts|ends_with(STR, STR2)->BOOL
                              #     - join(STR, ARR)->STR2
                              #     - keys(OBJ)->STR_ARR
                              #     - values(OBJ)->ARR
                              #     - length(OBJ|ARR|STR)->NUM
                              #     - reverse(ARR)->ARR2
                              #     - sort(ARR)->ARR2
                              #     - sort_by(ARR, FILTER)->ARR2
